 17 $wb.ui.layout.Stack = function() {
 18     this.elm().css({
 19         position:'relative'
 20     });
 21     var width = this.target().innerWidth();
 22     var height = this.target().innerHeight();
 23     var nodes = this.children();
 24     for(var i in nodes) {
 25         nodes[i].elm().css({
 26             position:'absolute',
 27             top:0,
 28             left:0,
 29             zIndex:i === 0 ? 10 : 1
 30         });
 31         nodes[i].elm().outerWidth(width);
 32         nodes[i].elm().outerHeight(height);
 33     }
 34 }

 38         nodes[i].elm().css({
 39             'float':'left'
 40         });
 41     }
 42 };

 57     var height = this.target().innerHeight();

 59     nodes[0].elm().outerHeight(height);

 64 $wb.ui.layout.GridBag = function() {
 65 
 66     var w = this.target().innerWidth();
 67     var h = this.target().innerHeight();
 68     var nodes = this.children();
 69     var others = [];
 70     if (nodes.length === 0)
 71         return;
 72 
 73     var last = null;
 74     for(var i in nodes) {
 75         var isLast = i == (nodes.length-1);
 76         if (isLast) {
 77             last = nodes[i].elm();
 78         } else {
 79             others.push(nodes[i].elm());
 80         }
 81     }
 82 
 83     var usedSize = $wb.utils.fullSize(others);
 84     last.outerHeight(h-usedSize.height);
 85     last.outerWidth(w);
 86 };

 90     /**
 91      * @description Base class for all widgets
 92      * @lends $wb.ui.Widget.prototype
 93      * @augments $wb.Class
 94      */

 96         __extends:[$wb.core.Events,$wb.core.Utils],
 97         /**
 98          * Base element
 99          * @private
100          */

102         /**
103          * Widgets unique id
104          * @private
105          */
106         _id:null,
107         /**
108          * Layout method. See {@link $wb.ui.layout}
109          * @private
110          */

113          * The template function.  See {@link $wb.ui.template}
114          * @private
115          */
116         _tmpl:null,
117         /**
118          * Child widgets
119          * @private
120          */
121         _children:[],
122         /**
123          * Target is the contained element which holds children. Defaults to {@link _elm}
124          * @private
125          */
126         _target:null,
127         /**
128          * Context menu
129          * @private
130          */
131         _context:null,
132         /**
133          * The provided options.
134          * @type Object
135          */
136         opts:{},
137         
138         /**
139         * @constructs
140         * @param {Object} opts 
141         * @param {Function} opts.tmpl template function
142         * @param {String} opts.id Id of the element
143         * @param {Function} opts.layout Layout function
144         */
145         __construct:function(opts) {
146             this.__super();
147 
148             this.require(opts,'tmpl');
149 
150             $.extend(true,this.opts,opts);
151 
152             this._makeElm();
153 
154             this._id = opts.id;
155 
156             if (this._id) {
157                 this.elm().attr('id',this._id);
158             }
159 
160             this._layoutMethod = opts.layout ? opts.layout : function() {};
161 
162             this.bind('resize',this._layout);
163         },
164         /**
165         * @private
166         */
167         _makeElm:function() {
168             this._tmpl = this.opts.tmpl;
169             var el = null;
170             if (typeof this._tmpl == 'function')
171                 el = $(this._tmpl());
172             else if (this._tmpl.tagName) {
173                 //dom element
174                 el = $(this._tmpl);
175             } else if (this._tmpl.css) {
176                 //jquery object
177                 if (this._tmpl.length > 0)
178                     el = $($(this._tmpl[0]).html());
179                 else
180                     throw "Empty jquery object received";
181             } else if (typeof this._tmpl == 'string') {
182                 //path
183                 el = $($(this._tmpl).html());
184                 if (el.length == 0) {
185                     throw "Empty jquery path received:"+this._tmpl;
186                 }
187             } else 
188                 throw "Invalid template argument provided:"+this._tmpl;
189 
190             if (this._elm) {
191                 this._elm.html(el.html());
192                 this._elm.unbind('click,keydown,keyup,mouseover,mousedown');
193             } else {
194                 this._elm = el;
195                 if (this.opts["class"]) {
196                     el.addClass(this.opts["class"]);
197                 }
198                 this._elm.widget(this);
199             }
200 
201             if (this.opts.target) {
202                 this._target = this._elm.find(this.opts.target);
203             } else {
204                 this._target = this._elm;
205             }
206         },
207         /**
208          * Add child 
209          * @params {$wb.ui.Widget} child
210          * @returns {$wb.ui.Widget} itself
211          */
212         add:function(child) {
213             this.children().push(child);
214             return this;
215         },
216         /**
217          * Get all child widgets of this widget
218          * @returns {$wb.ui.Widget[]} children
219          */
220         children:function() {
221             return this._children;
222         },
223         /**
224          * Set child index ix to child - replacing/removing any child at that index
225          * @params {int} ix Child index
226          * @params {$wb.ui.Widget} child
227          * @returns {$wb.ui.Widget} itself
228          */
229         set:function(ix,child) {
230             this.remove(ix);
231             this._children[ix] = child;
232             return this;
233         },
234         /**
235          *Get index of child - returns -1 if not found
236          * @returns {int} index
237          */
238         indexOf:function(child) {
239             for(var i = 0; i < this._children.length;i++) {
240                 if (this._children[i] == child)
241                     return i;
242             }
243             return -1;
244         },
245         /**
246          * Detach child by index or object
247          * @returns {$wb.ui.Widget} removed widget
248          */
249         remove:function(ix) {
250             if (typeof ix != 'number') {
251                 ix = this.indexOf(ix);
252             }
253 
254             if (ix > -1 && this._children[ix]) {
255                 var out = this._children[ix];
256                 this._children.splice(ix,1);
257                 out.detach();
258                 return out;
259             }
260 
261             return null;
262         },
263         /**
264          * Detach all children
265          * @returns {$wb.ui.Widget} itself
266          */
267         clear:function() {
268             while(this._children.length > 0) {
269                 var child = this._children.pop();
270                 child.detach();
271             }
272             return this;
273         },
274         /**
275          * Show widget
276          * @returns {$wb.ui.Widget} itself
277          */
278         show:function() {
279             this.elm().show();
280             this.trigger('show');
281             return this;
282         },
283         /**
284          * Hide widget
285          * @returns {$wb.ui.Widget} itself
286          */
287         hide:function() {
288             this.elm().hide();
289             this.trigger('hide');
290             return this;
291         },
292 
293         /**
294          * Completely erase this widget from memory
295          * @paras {Boolean} recurse If true - destroys all children too (Defaults to just detaching them)
296          */
297         destroy:function(recurse) {
298 
299             while(this._children.length > 0) {
300                 var child = this._children.pop();
301                 if (recurse) 
302                     child.destroy();
303                 else
304                     child.detach();
305             }
306             if (this.parent())
307                 this.parent().remove(this);
308             delete this.opts;
309             this.elm().detach();
310             delete this;
311         },
312         /**
313          * Detach this widget
314          * @returns {$wb.ui.Widget}
315          */
316         detach:function() {
317             this.elm().detach();
318             return this;
319         },
320         /**
321          * Get parent widget
322          * @returns {$wb.ui.Widget} parent widget
323          */
324         parent:function() {
325             return this.elm().parent().closest('.-wb-state-widget').widget();
326         },
327         /**
328          * Set or Get html
329          */
330         html: function(html) {
331             return this.target().html(html);
332         },
333         /**
334          * find DOM element within widget
335          * @returns {jQueryElement}
336          */
337         find: function(path) {
338             return this.elm().find(path);
339         },
340         /**
341          * Get base element
342          * @returns {jQueryElement}
343          */
344         elm: function() {
345             return this._elm;
346         },
347         /**
348          * Get target - which is the element within this widget that holds children.
349          * @returns {jQueryElement}
350          */
351         target:function() {
352             return this._target;
353         },
354         /**
355          * Render widget
356          * @param {jQueryElement} [container] Optional element in which to append this widget
357          * @returns {jQueryElement} the base element of this widget
358          */
359         render: function(container) {
360             if (this._paint() === false) 
361                 return false;
362             this._place(container);
363 
364             this._layout();
365 
366             this._renderChildren();
367 
368             this._layout();
369 
370             this.trigger('render');
371             return this.elm();
372         },
373         /**
374          * Set context menu
375          * @param {$wb.ui.Context} w context menu
376          */
377         setContextMenu:function(w) {
378             var ccontext_id = '-wb-state-current-context';
379             var elm = w.elm();
380 
381             var onHide = function(evt) {
382                 evt.stopPropagation();
383                 elm.detach();
384                 $('body').unbind('click',onHide);
385                 elm.unbind('click',onHide);
386                 elm.unbind('contextmenu',onHide);
387             };
388 
389             var onContext = function(evt) {
390                 evt.preventDefault();
391                 evt.stopPropagation();
392 
393                 //Remove all others
394                 $('.'+ccontext_id).detach();
395 
396                 $('body').append(elm);
397 
398                 elm.css({
399                     position:'absolute',
400                     left:evt.pageX,
401                     top:evt.pageY,
402                     zIndex:9999
403                 });
404 
405                 var source = $(evt.target).widget();
406                 if (typeof source == 'undefined') {
407                     console.log("Could not find widget for element");
408                     return;
409                 }
410                 elm.addClass(ccontext_id);
411                 w.source(source);
412                 w.render();
413 
414                 elm.bindOnce('click',onHide);
415                 $('body').bindOnce('click',onHide);
416                 $('body').bindOnce('contextmenu',onHide);
417             };
418 
419             var onExit = function() {
420                 $('.'+ccontext_id).detach();
421             };
422 
423 
424             this.bind('paint',function() {
425                 this.elm().bindOnce('contextmenu',onContext);
426                 this.elm().bindOnce('click',onExit);
427             });
428             return this;
429         },
430         /**
431         * @private
432         */
433         _paint: function() {
434             if (this.trigger('before-paint') === false)
435                 return false;
436             for(var i = 0; i < this.children().length;i++) {
437                 //Only add elements not already added
438                 if (!this.target().contains(this._children[i].elm()))
439                     this.target().append(this._children[i].elm());
440             }
441             if (this.trigger('paint') === false)
442                 return false;
443             return true;
444         },
445         /**
446         * @private
447         */
448         _place: function(container) {
449             if (container) {
450                 $(container).append(this.elm());
451             }
452         },
453         /**
454         * @private
455         */
456         _layout: function() {
457             this.trigger('beforelayout');
458             if (this._layoutMethod) {
459                 this._layoutMethod.apply(this);
460             }
461             for(var i in this._children) {
462                 var child = this._children[i];
463                 child._layout();
464             }
465             this.trigger('afterlayout');
466         },
467         /**
468         * @private
469         */
470         _renderChildren: function() {
471             this.trigger('beforerenderchildren');
472             for(var i in this._children) {
473                 this._children[i].render();
474             }
475         },
476         /**
477         * @private
478         */
479         _resize: function() {
480             this._layout.apply(this);
481             for(var i in this._children) {
482                 var child = this._children[i];
483                 child._resize();
484             }
485         }
486     }
