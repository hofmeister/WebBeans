 12 $wb.utils = {
 13     _typeResolvers:[
 14         function(v) {
 15             if ($wb.utils.isClass(v))
 16                 return v.clz;
 17             return null;
 18         },
 19         function(v) {
 20             if ($wb.utils.isA(v,'Model'))
 21                 return "Model/"+v.getType();
 22             return null;
 23         },
 24         function(v) {
 25             return $.type(v);
 26         }
 27     ],
 28     /**
 29      * Get value using path on object
 30      * @param {Object} obj the object
 31      * @param {String} path the path
 32      * @type Object
 33      * @example
 34      * 
 35      * var obj = {hello:{world:"test"}}
 36      * var val = $wb.utils.GetValue(obj,"hello.world");
 37      * console.log(val); //Emits "test"
 38      */
 39     GetValue:function(obj,path) {
 40         var parts = path.split('.');
 41         var cur = obj;
 42         for(var i in parts) {
 43             var p = parts[i];
 44             cur = cur[p];
 45             if (!cur)
 46                 break;
 47         }
 48         return cur;
 49     },
 50     /**
 51      * Sets value using path on object
 52      * @param {Object} obj the object
 53      * @param {String} path the path
 54      * @param {Object} value
 55      * @example
 56      * 
 57      * var obj = {hello:{world:"test"}}
 58      * var val = $wb.utils.SetValue(obj,"hello.world","hi");
 59      * console.log(obj.hello.world); //Emits "hi"
 60      */
 61     SetValue:function(obj,path,value) {
 62         var parts = path.split('.');
 63         var cur = obj;
 64         for(var i in parts) {
 65             var p = parts[i];
 66             if (i == (parts.length-1))
 67                 break;
 68             cur = cur[p];
 69             if (!cur)
 70                 break;
 71         }
 72         if (p && cur)
 73             cur[p] = value;
 74     },
 75     /**
 76      * Make an element full screen - optionally listening for window resize events
 77      * @param {DOMNode|String} elm the element (passed to jQuery)
 78      * @param {boolean} listenForResize if true - listen for resize events
 79      */
 80     makeFullScreen:function(elm,listenForResize) {
 81         var resize = function() {
 82             var w = $(window).width();
 83             var h = $(window).height();
 84             $(elm).width(w);
 85             $(elm).height(h);
 86             elm.trigger('resize');
 87         };
 88         if (listenForResize) {
 89             $(window).bind('resize',resize);
 90         }
 91         resize();
 92     },
 93     /**
 94      * Get full size including padding, borders and margin for element
 95      * @param {DOMNode|String} elm the element (passed to jQuery)
 96      * @type Object
 97      */
 98     fullSize:function(elm) {
 99         var out = {
100             width:0,
101             height:0
102         };
103         $(elm).each(function() {
104             var el = $(this);
105             out.width += el.width()
106             +parseInt(el.css('padding-left'))
107             +parseInt(el.css('padding-right'))
108             +parseInt(el.css('margin-left'))
109             +parseInt(el.css('margin-right'))
110             +parseInt(el.css('border-left-width'))
111             +parseInt(el.css('border-right-width'));
112             out.height += el.height() 
113             +parseInt(el.css('padding-top'))
114             +parseInt(el.css('padding-bottom'))
115             +parseInt(el.css('margin-top'))
116             +parseInt(el.css('margin-bottom'))
117             +parseInt(el.css('border-top-width'))
118             +parseInt(el.css('border-bottom-width'));
119         })
120         return out;
121     },
122     
123     /**
124      * make element fill available space in parent
125      * @param {DOMNode|String} elm the element (passed to jQuery)
126      * @param {Boolean} listenForResize listen for resize events on parent
127      * @type Object
128      */
129     fillContainer:function(elm,listenForResize) {
130         var parent = $(elm).parent();
131         
132         var resize = function() {
133             var w = parent.width();
134             var h = parent.height();
135             var others = parent.children().not(elm);
136             var usedSize = $wb.utils.fullSize(others);
137             elm.height(h-usedSize.height);
138             elm.width(w);
139         };
140         if (listenForResize) {
141             parent.bind('resize',resize);
142         }
143         resize();
144     },

157     /**
158      * Get class name of instance (created with $wb.Class)
159      * @param {Object} obj The instance
160      * @type String
161      */
162     getClass:function(obj) {
163         if (this.isClass(obj))
164             return obj._clz;
165         return null;
166     },
167     /**
168      * Determine if an object is a class (created with $wb.Class)
169      * @param {Object} obj The instance
170      * @type Boolean
171      */
172     isClass:function(obj) {
173         return obj && (typeof obj._clz != 'undefined');
174     },
175     /**
176      * Determine if an object is an instance of class. Does not yet check for inheritance.
177      * @param {Object} obj The instance
178      * @param {Class} clz The class
179      * @type Boolean
180      */
181     isA:function(obj,clz) {
182         if (!this.isClass(obj)) 
183             return false;
184         return obj._clz == clz;
185     },
186     /**
187      * Type resolving of value
188      * @param {Object} v value
189      * @type String
190      */
191     type:function(v) {
192         if (v) {
193             for(var i = 0; i < this._typeResolvers.length;i++) {
194                 var func = this._typeResolvers[i];
195                 var type = func(v);
196                 if (type != null)
197                     return type;
198             }
199         }
200         return "undefined";
201     }

207     jQuery.fn.widget = function(widget) {
208         var clz = '-wb-state-widget';
209         var elm = $(this);
210         if (widget) {
211             elm .addClass(clz);
212             return elm.data('__widget',widget);
213         } else {
214             if (elm.is('.'+clz))
215                 return elm.data('__widget');
216             return elm.closest('.'+clz).data('__widget');
217         }
218     };

373     jQuery.fn.keyboardNavigation = function() {
374         var elm = $(this);
375         
376         if (elm.children('input.wb-keynav-input').length > 0) 
377             return;
378         
379         var input = $('<input type="text" class="wb-keynav-input" style="position:absolute;left:-9999px" />')
380         elm.append(input);
381         
382         elm.bind('mousedown',function(evt) {
383             evt.preventDefault();
384             $(this).mouseup();
385             $(this).click();
386         })
387         elm.bind('click mousedown',function(evt) {
388             evt.stopPropagation();
389             input.focus();
390         });
391         input.focus(function() {
392             elm.addClass('wb-focus');
393         });
394         input.blur(function(evt) {
395             elm.removeClass('wb-focus');
396         });
397         
398     };
